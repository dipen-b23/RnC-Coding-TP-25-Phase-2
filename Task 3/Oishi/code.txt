// Motor pins
#define IN1 2
#define IN2 3
#define IN3 4
#define IN4 5

// Enable pins
#define EN1 9
#define EN2 10

// Buttons
#define FORWARD_BTN 6
#define BACKWARD_BTN 7
#define LEFT_BTN 8
#define RIGHT_BTN 12

// Potentiometer
#define POT A0

void setup() {
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);

  pinMode(EN1, OUTPUT);
  pinMode(EN2, OUTPUT);

  pinMode(FORWARD_BTN, INPUT_PULLUP);
  pinMode(BACKWARD_BTN, INPUT_PULLUP);
  pinMode(LEFT_BTN, INPUT_PULLUP);
  pinMode(RIGHT_BTN, INPUT_PULLUP);
}

void loop() {

  int speed = analogRead(POT);
  speed = map(speed, 0, 1023, 0, 255);

  analogWrite(EN1, speed);
  analogWrite(EN2, speed);

  if (digitalRead(FORWARD_BTN) == LOW) {
    digitalWrite(IN1, HIGH);
    digitalWrite(IN2, LOW);
    digitalWrite(IN3, HIGH);
    digitalWrite(IN4, LOW);
  }

  else if (digitalRead(BACKWARD_BTN) == LOW) {
    digitalWrite(IN1, LOW);
    digitalWrite(IN2, HIGH);
    digitalWrite(IN3, LOW);
    digitalWrite(IN4, HIGH);
  }

  else if (digitalRead(LEFT_BTN) == LOW) {
    digitalWrite(IN1, LOW);
    digitalWrite(IN2, LOW);
    digitalWrite(IN3, HIGH);
    digitalWrite(IN4, LOW);
  }

  else if (digitalRead(RIGHT_BTN) == LOW) {
    digitalWrite(IN1, HIGH);
    digitalWrite(IN2, LOW);
    digitalWrite(IN3, LOW);
    digitalWrite(IN4, LOW);
  }

  else {
    digitalWrite(IN1, LOW);
    digitalWrite(IN2, LOW);
    digitalWrite(IN3, LOW);
    digitalWrite(IN4, LOW);
  }
}
